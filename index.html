const SUBS = ["surrealmemes", "hmm", "skamtebord", "blursedimages", "me_irl", "comedyheaven"]; const CORS_HEADERS = { "Access-Control-Allow-Origin": "*", "Access-Control-Allow-Headers": "x-app-token, apikey, authorization, content-type", "Access-Control-Allow-Methods": "GET, OPTIONS", "Content-Type": "application/json", }; Deno.serve(async (req) => { if (req.method === "OPTIONS") { return new Response("ok", { headers: CORS_HEADERS }); } try { const sub = SUBS[Math.floor(Math.random() * SUBS.length)]; const res = await fetch( `https://www.reddit.com/r/${sub}/hot.json?limit=50&raw_json=1`, { headers: { "User-Agent": "OpenBrain/1.0" } } ); if (!res.ok) throw new Error(`Reddit responded ${res.status}`); const data = await res.json(); const posts = (data.data?.children || []) .map((c) => c.data) .filter( (p) => !p.over_18 && !p.stickied && p.post_hint === "image" && p.url && /\.(jpg|jpeg|png|gif|webp)(\?.*)?$/i.test(p.url) ) .map((p) => ({ url: p.url, title: p.title, sub: "r/" + p.subreddit })); if (!posts.length) { return new Response(JSON.stringify({ error: "no_images", sub }), { headers: CORS_HEADERS, }); } const pick = posts[Math.floor(Math.random() * posts.length)]; return new Response(JSON.stringify(pick), { headers: CORS_HEADERS }); } catch (err) { return new Response(JSON.stringify({ error: String(err.message) }), { status: 500, headers: CORS_HEADERS, }); } });
